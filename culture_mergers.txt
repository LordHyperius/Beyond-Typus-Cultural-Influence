merge_rhomaio = {
 every_province = {
	limit = {
		OR = {
			culture = greek
			culture = pontic_greek
			culture = cappadocian
			}
		}
		change_culture = rhomaio
	}
	every_country = {
		limit = {
			OR = {
			primary_culture = greek
			primary_culture = pontic_greek
			primary_culture = cappadocian
			}
		}
		change_primary_culture = rhomaio
	}
	every_country = {
		limit = {
			OR = {
				has_accepted_culture = greek
				has_accepted_culture = pontic_greek
				has_accepted_culture = cappadocian
			}
		}
		remove_accepted_culture = greek
		remove_accepted_culture = pontic_greek
		remove_accepted_culture = cappadocian
		add_accepted_culture = rhomaio
	}
	every_country = {
		limit = {
			OR = {
				ruler_culture = greek
				ruler_culture = pontic_greek
				ruler_culture = cappadocian
			}
		}
		set_ruler_culture = rhomaio
	}
	every_country = {
		limit = {
			OR = {
				heir_culture = greek
				heir_culture = pontic_greek
				heir_culture = cappadocian
			}
		}
		set_heir_culture = rhomaio
	}
	every_country = {
		limit = {
			OR = {
			consort_culture = greek
			consort_culture = pontic_greek
			consort_culture = cappadocian
			}
		}
		set_consort_culture = rhomaio
	}
}
